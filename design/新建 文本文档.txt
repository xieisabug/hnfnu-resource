CREATE 
    ALGORITHM = UNDEFINED 
    DEFINER = `root`@`localhost` 
    SQL SECURITY DEFINER
VIEW `v_menu_menu` AS
    select 
        `t1`.`id` AS `id`,
        `t1`.`name` AS `name`,
        `t1`.`url` AS `url`,
        `t1`.`functionIdList` AS `functionIdList`,
        `t1`.`icon` AS `icon`,
        `t1`.`parentId` AS `parentId`,
        `t2`.`name` AS `parentName`
    from
        (`s_menu` `t1`
        join `s_menu` `t2` ON ((`t2`.`id` = `t1`.`parentId`)))
